    function pickIndex(key, len) {

      const last = Number(sessionStorage.getItem(key) ?? -1);

      if (len <= 1) return 0;

      let i = Math.floor(Math.random() * len);

      if (i === last) i = (i + 1) % len;

      sessionStorage.setItem(key, i);

      return i;

    }
